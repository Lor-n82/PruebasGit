git init => Iniciar seguimiento
git status => Ver estado
git add => Añade al area de Staging
git commit -m "mensaje" => Añade al repositorio local
git log --online => muestra el log
git reset --hard CodigoCommit => Volver al commit deseado
git commit -am => Hace un add antes de commitar los cambios
git commit --amend => Abre un editor VIM
    :i => Editar
    ESC => Salir del modo
    :wq => guardar y salir

git tag nombretag -m "mensaje" => Añade una etiqueta
git push --tags => Subit tags
